function [r,u,t,a] = ruta(fo,f,b,p)
c = 3*10^8; r = c/2/b;
u = c/2/f; t = f*100/b; a = t*p;
disp(['range resolution: ', num2str(r), ' m']);
disp(['maximum unambiguous range: ', num2str(u),...
' m']);
disp(['transmit duty cycle: ', num2str(t), ' %']);
disp(['average power in the system: ', num2str(a), ...
' W']);
end
